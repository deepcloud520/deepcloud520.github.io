        def story0_home_awake(eventrunner):
            # 加载 EntityEvilNeuro
            aris.get_rooms().update(aris.makerooms('rooms/'+resmanager.DefResourceDomain.get_resource('rooms.story0')['resource']))
            aris.get_rooms()[resmanager.DefResourceDomain.get_resource('rooms.story0')['start_room']].build(get_world().engine)
            player = get_entity_bydefname(get_world().engine.livings_line,defname='evilneuro')
            acr = AirisCoreRunable(player)
            get_MCO().append(acr)
            eventrunner.Runables[RING0].append(acr)
            
            self.framemanager.hide_all()
